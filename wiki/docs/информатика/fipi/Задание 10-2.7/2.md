function checkButtonClick(form, setanswer, panel) {
	form = document.getElementById(form);
	setanswer(form);
    if (!request.inProcess) {
        request.open('POST', 'solve.php', true);
        request.onreadystatechange = checkSolutionDone;
        request.inProcess = true;
        request.panel = panel;

        var formData = new FormData(form);
        formData.append("ajax", "1");
        formData.append("proj", "74676951F093A0754D74F2D6E7955F06")

		if (browser.isIE)
			request.qst = form.guid.value;
		else
	        request.qst = formData.get("guid");

        request.send(formData);

        panel.getElementsByClassName('task-status')[0].innerHTML = "<img class='loader' src='img/loading_spinner.gif'>";
    }
}

function checkSolutionDone() {
    if (request.readyState==4)  {
        request.inProcess = false;
        answer = "0";
        if (request.status==200) {
            answer = request.responseText;
        }
        if (answer!='1' && answer!='2' && answer!='3') {
        	answer = "0";
        }
        qst = request.qst;

        if (answer == '0' || answer == '2')
    		solvedQst.delete(qst);
        else
            solvedQst.add(qst);
        localStorage.setItem("OS:74676951F093A0754D74F2D6E7955F06:SOLVED", setToStr(solvedQst));
        setQSolved(request.panel, answer);
    }
}

	if (typeof(block) === 'string')
	    block = document.getElementById(block);
	var statusNode = block.getElementsByClassName('task-status')[0];
    statusNode.className= "task-status task-status-" + answer ;
    if (answer=='1')
        statusNode.innerHTML = "РЕШЕНО";
    else if (answer=='2')
        statusNode.innerHTML = "НЕВЕРНО";
    else if (answer=='3')
        statusNode.innerHTML = "ВЕРНО";
    else
        statusNode.innerHTML = "НЕ РЕШЕНО";
}
